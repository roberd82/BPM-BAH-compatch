						bpm_leader_is_socialist = yes

				}
			}
		}
		NOT = { has_variable = bah_controlled_hungarian_government }

			}

			name = modifier_hungary_curtailed
			months = normal_modifier_time
		}
	}

	ai_chance = {
		base = 1
		modifier = {
			trigger = { authority > 300 }
			add = 10
		}		
	}
}
