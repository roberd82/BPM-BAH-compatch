namespace = bpm_bah_events

# After revolution content
bpm_bah_events.1 = {
	type = country_event

	title = bpm_bah_events.1.t
	desc = bpm_bah_events.1.d
	flavor = bpm_bah_events.1.f

	event_image = {
		video = "gfx/event_pictures/unspecific_ruler_speaking_to_people.bk2"
	}

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/leader_speaking_to_a_group_of_people"

	duration = 1

	dlc = dlc_bpm

	trigger = {
		# triggered by je_hungary_resistance or je_transylvania_resistance
	}

	option = {
		name = bpm_bah_events.1.a # stay monarchy for now
		if = {
			limit = {
				OR = {
					this = c:HUN
					AND = {
						OR = {
							this = c:TRS
							AND = {
								this = c:CRO
								c:CRO = {
									is_direct_subject_of = c:HUN
								}
							}
						}
						c:HUN = {
							has_law = law_type:law_monarchy
						}
					}
				}
			}
			if = {
				limit = {
					OR = {
						this = c:TRS
						this = c:CRO
					}
					c:HUN = {
						has_law = law_type:law_monarchy
					}
				}
				default_option = yes
			}
			trigger_event = { 
				id = bpm_bah_events.2
				days = 2
				popup = yes
			}
		}
	}

	option = {
		trigger = {
			OR = {
				this = c:HUN
				AND = {
					this = c:CRO
					c:CRO = {
						NOT = { is_direct_subject_of = c:HUN }
					}
				}
			}
		}
		name = bpm_bah_events.1.b # become presidential republic
		default_option = yes
		if = {
			limit = {
				this = c:HUN
			}
			trigger_event = {
				id = bpm_bah_events.2
				days = 2
				popup = yes
			}
		}
		activate_law = law_type:law_presidential_republic
		add_journal_entry = {
			type = je_abolish_monarchy
		}
	}

	option = {
		name = bpm_bah_events.1.c # become parliamentary republic
		if = {
			limit = {
				OR = {
					this = c:TRS
					AND = {
						this = c:CRO
						c:CRO = {
							is_direct_subject_of = c:HUN
						}
					}
				}
				c:HUN = {
					NOT = { has_law = law_type:law_monarchy }
				}
			}
			default_option = yes
		}
		activate_law = law_type:law_parliamentary_republic
		add_journal_entry = {
			type = je_abolish_monarchy
		}
	}
}

bpm_bah_events.2 = {
	type = country_event

	title = bpm_bah_events.2.t
	desc = bpm_bah_events.2.d
	flavor = bpm_bah_events.2.f

	event_image = {
		video = "gfx/event_pictures/unspecific_ruler_speaking_to_people.bk2"
	}

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/leader_speaking_to_a_group_of_people"

	duration = 1

	dlc = dlc_bpm

	trigger = {
		# triggered by bpm_bah_events.1.a and bpm_bah_events.1.b
	}

	option = {
		name = bpm_bah_events.2.a # keep the temp ruler
		default_option = yes
	}

	option = {
		trigger = {
			this = c:HUN
		}
		name = bpm_bah_events.2.b # elect Sz√©chenyi as king/first president

		ig:ig_industrialists = {
			leader = {
				set_character_as_ruler = yes
			}
		}
	}

	option = {
		trigger = {
			this = c:HUN
			has_law = law_type:law_presidential_republic
		}
		name = bpm_bah_events.2.c # elect Kossuth as first president

		ig:ig_radicals = {
			leader = {
				set_character_as_ruler = yes
			}
		}
	}

	option = {
		trigger = {
			OR = {
				this = c:TRS
				AND = {
					this = c:CRO
					c:CRO = {
						is_direct_subject_of = c:HUN
					}
				}
			}
			has_law = law_type:law_monarchy
			c:HUN = {
				has_law = law_type:law_monarchy
			}
		}
		
		name = bpm_bah_events.2.e # pu with HUN

		if = {
			limit = {
				this = c:TRS
			}
			c:HUN = {
				create_diplomatic_pact = {
					country = c:TRS
					type = personal_union
				}
			}
		}
		if = {
			limit = {
				this = c:CRO
			}
			c:HUN = {
				create_diplomatic_pact = {
					country = c:CRO
					type = personal_union
				}
			}
		}
	}
}
