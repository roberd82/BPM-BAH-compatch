namespace = bpm_bah_events

bpm_bah_events.1 = {
	type = country_event

	title = bpm_bah_events.1.t
	desc = bpm_bah_events.1.d
	flavor = bpm_bah_events.1.f

	event_image = {
		video = "gfx/event_pictures/unspecific_ruler_speaking_to_people.bk2"
	}

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/leader_speaking_to_a_group_of_people"

	duration = 1

	dlc = dlc_bpm

	trigger = {
		# triggered by je_hungary_resistance or je_transylvania_resistance
	}

	option = {
		name = bpm_bah_events.1.a # stay monarchy for now
		if = {
			limit = {
				this = c:TRS
				c:HUN = {
					has_law = law_type:law_monarchy
				}
			}
			default_option = yes
		}
		trigger_event = { 
			id = bpm_bah_events.2
		}
	}

	option = {
		trigger = {
			this = c:HUN
		}
		name = bpm_bah_events.1.b # become presidential republic
		default_option = yes
		trigger_event = { 
			id = bpm_bah_events.2
		}
		activate_law = law_type:law_presidential_republic
	}

	option = {
		name = bpm_bah_events.1.c # become parliamentary republic
		if = {
			limit = {
				this = c:TRS
				c:HUN = {
					NOT = { has_law = law_type:law_monarchy }
				}
			}
			default_option = yes
		}
		activate_law = law_type:law_parliamentary_republic
	}
}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/leader_speaking_to_a_group_of_people"

	duration = 1

	dlc = dlc_bpm

	trigger = {
		# triggered by bpm_bah_events.1.a and bpm_bah_events.1.b
	}

	option = {
		name = bpm_bah_events.2.a # keep the temp ruler

	}

	option = {
		trigger = {
			this = c:HUN
		}
		name = bpm_bah_events.2.b # elect Sz√©chenyi as king/president

		ig:ig_industrialists = {
			leader = {
				set_character_as_ruler = yes
			}
		}
	}

	option = {
		trigger = {
			this = c:HUN
			has_law = law_type:law_presidential_republic
		}
		name = bpm_bah_events.2.c # elect Kossuth as president

		ig:ig_radicals = {
			leader = {
				set_character_as_ruler = yes
			}
		}
	}

	option = {
		trigger = {
			this = c:TRS
			has_law = law_type:law_monarchy
		}
		name = bpm_bah_events.2.d # TRS pu with HUN
		c:HUN = {
			create_diplomatic_pact = {
				country = c:TRS
				type = personal_union
			}
		}
	}
}
