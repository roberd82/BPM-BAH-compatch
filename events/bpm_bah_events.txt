namespace = bpm_bah_events

# After revolution content
bpm_bah_events.1 = {
	type = country_event

	title = bpm_bah_events.1.t
	desc = bpm_bah_events.1.d
	flavor = bpm_bah_events.1.f

	event_image = {
		video = "gfx/event_pictures/unspecific_ruler_speaking_to_people.bk2"
	}

	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/leader_speaking_to_a_group_of_people"

	duration = 1

	dlc = dlc_bpm

	trigger = {
		# triggered by je_hungary_resistance or je_transylvania_resistance
	}

	option = {
		name = bpm_bah_events.1.a # stay monarchy for now
		if = {
			limit = {
				OR = {
					this = c:HUN
					AND = {
						this = c:TRS
						c:HUN = {
							has_law = law_type:law_monarchy
						}
					}
				}
			}
			if = {
				limit = {
					this = c:TRS
					c:HUN = {
						has_law = law_type:law_monarchy
					}
				}
				default_option = yes
			}
			trigger_event = { 
				id = bpm_bah_events.2
				days = 2
				popup = yes
			}
		}
	}

	option = {
		trigger = {
			this = c:HUN
		}
		name = bpm_bah_events.1.b # become presidential republic
		default_option = yes
		trigger_event = { 
			id = bpm_bah_events.2
			days = 2
			popup = yes 
		}
		activate_law = law_type:law_presidential_republic
		add_journal_entry = {
			type = je_abolish_monarchy
		}
	}

	option = {
		name = bpm_bah_events.1.c # become parliamentary republic
		if = {
			limit = {
				this = c:TRS
				c:HUN = {
					NOT = { has_law = law_type:law_monarchy }
				}
			}
			default_option = yes
		}
		activate_law = law_type:law_parliamentary_republic
		add_journal_entry = {
			type = je_abolish_monarchy
		}
	}
}
